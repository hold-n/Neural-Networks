def print_stat(df, label):
    print(label, len(df), 'files with', (df['species'] == 'dog').sum(), 'dogs')
    

def create_df(files):
    return pd.DataFrame(
        [(str(f), 'dog' if f.stem.startswith('dog') else 'cat') for f in files], 
        columns=['filename', 'species']
    )

img_shape=(100, 100, 3)

def flow(df, gen=None, input_shape=img_shape):
    if gen is None:
        gen = keras.preprocessing.image.ImageDataGenerator(rescale=1./255)
    return gen.flow_from_dataframe(
        df, 
        directory='.', 
        batch_size=32,
        target_size=input_shape[:2], 
        x_col='filename', 
        y_col='species', 
        class_mode='binary'
    )

def dense_layers(n=70):
    return keras.Sequential([
        keras.layers.Flatten(),
        keras.layers.Dropout(0.2),
        keras.layers.Dense(n, activation='relu'),
        keras.layers.Dropout(0.2),
        keras.layers.Dense(1)
    ])

def create_model():
    return keras.Sequential([
        keras.layers.Conv2D(filters=24, kernel_size=(3, 3), activation='relu', input_shape=img_shape),
        keras.layers.MaxPooling2D(pool_size=2),
        keras.layers.Conv2D(filters=48, kernel_size=(5, 5), activation='relu'),
        keras.layers.MaxPooling2D(pool_size=2),
        keras.layers.Conv2D(filters=96, kernel_size=(7, 7), activation='relu'),
        keras.layers.MaxPooling2D(pool_size=2),
        dense_layers()
    ])

def compile_model(model):
    model.compile(
        optimizer='adam', 
        loss=tf.keras.losses.BinaryCrossentropy(from_logits=True), 
        metrics=['accuracy']
    )

    weights='imagenet', 
    include_top=False

x = dense_layers(128)(x)
inception_model = keras.Model(inputs=inception_model_raw.input, outputs=x)

    layer.trainable = False
